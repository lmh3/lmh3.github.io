S1 = 1 - 1 + 1 - 1 + 1 - 1 ...

S2 = 1 - 2 + 3 - 4 + 5 - 6 ...

S = 1 + 2 + 3 + 4 + 5 + 6 ...

S1 = 1/2

2S2 = [1 - 2 + 3 - 4 + 5 - 6 ...] + [1 - 2 + 3 - 4 + 5 - 6 ...]

    = 1 - 1 + 1 - 1 + 1 - 1 ...
    
S2 = 1/4

S - S2 = [1 + 2 + 3 + 4 + 5 + 6 ...] - [1 - 2 + 3 - 4 + 5 - 6 ...]

		= 2(2 + 4 + 6 + 8 ...)
		
		= 4(1 + 2 + 3 + 4 + 5 + 6 ...)
		
		= 4S
		
S = -1/12
